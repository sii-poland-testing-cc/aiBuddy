    1. Create `frontend/lib/apiFetch.ts` (per D-12):
       ```typescript
       const API_BASE = process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:8000";

       /**
        * Thin wrapper around fetch() that:
        * - Prepends API_BASE to relative paths
        * - Always sends credentials: "include" for httpOnly cookie transport
        *
        * Usage: apiFetch("/api/projects") instead of fetch(`${API_BASE}/api/projects`)
        */
       export function apiFetch(path: string, init?: RequestInit): Promise<Response> {
         return fetch(`${API_BASE}${path}`, {
           ...init,
           credentials: "include",
         });
       }

       /** Re-export for hooks that need the base URL directly (e.g., SSE URL construction). */
       export { API_BASE };
       ```
       IMPORTANT: Spread `...init` first, then `credentials: "include"` after — so callers cannot accidentally override credentials.

    1. Create `frontend/middleware.ts` (per D-10):
       ```typescript
       import { NextRequest, NextResponse } from "next/server";

       export function middleware(request: NextRequest) {
         const token = request.cookies.get("access_token");

         const isAuthPage =
           request.nextUrl.pathname.startsWith("/login") ||
           request.nextUrl.pathname.startsWith("/register");

         // If authenticated and on auth page, redirect to home
         if (token && isAuthPage) {
           return NextResponse.redirect(new URL("/", request.url));
         }

         // If not authenticated and on protected page, redirect to login
         if (!token && !isAuthPage) {
           return NextResponse.redirect(new URL("/login", request.url));
         }

         return NextResponse.next();
       }

       export const config = {
         matcher: [
           /*
            * Match all request paths EXCEPT:
            * - _next/static (static files)
            * - _next/image (image optimization)
            * - favicon.ico
            */
           "/((?!_next/static|_next/image|favicon\\.ico).*)",
         ],
       };
       ```
       CRITICAL: Matcher excludes `_next` and `favicon.ico` only. Auth page detection uses pathname check inside the function, NOT the matcher (per Pitfall 3 from RESEARCH.md — route group `(auth)` is invisible in URL, so `/login` is the URL path, not `/(auth)/login`).

               router.push("/");
             }

             setError("Blad serwera. Sprobuj ponownie za chwile.");

             setLoading(false);
           }

         return (

                   <input
                     id="email"
                     type="email"
                     value={email}
                     onChange={(e) => setEmail(e.target.value)}
                     placeholder="you@example.com"
                     required
                     className="w-full bg-buddy-elevated border border-buddy-border rounded-xl px-4 py-3 text-sm text-buddy-text placeholder:text-buddy-text-faint focus:outline-none focus:border-buddy-gold transition-colors duration-150"

                     placeholder="••••••••"
                     required
                     className="w-full bg-buddy-elevated border border-buddy-border rounded-xl px-4 py-3 text-sm text-buddy-text placeholder:text-buddy-text-faint focus:outline-none focus:border-buddy-gold transition-colors duration-150"
                   />

                     {error}
                   </div>

                   disabled={loading}
