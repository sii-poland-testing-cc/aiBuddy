```python
from app.core.rbac import can_user
from app.core.config import settings

@router.post("/stream")
async def chat_stream(
    req: ChatRequest,
    request: Request,
    current_user = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    if settings.ENFORCE_AUTH:
        if not isinstance(current_user, AnonymousUser):
            allowed = await can_user(current_user.id, "write", "project", req.project_id, db, request)
            if not allowed:
                raise HTTPException(status_code=403, detail="Brak uprawnien do uruchomienia tej operacji.")
    return StreamingResponse(...)
```

**C. Bootstrap endpoint in auth.py (per D-10):**

Add to `backend/app/api/routes/auth.py`:

```python
from sqlalchemy import func

class BootstrapRequest(BaseModel):
    email: str
    password: str

@router.post("/bootstrap", status_code=201)
async def bootstrap_superadmin(body: BootstrapRequest, db: AsyncSession = Depends(get_db)):
    """
    D-10: Create the first superadmin when users table is empty.
    Open endpoint (no auth required) — auth.py router has no router-level guard.
    Returns 409 if any user already exists.
    """
    count = (await db.execute(select(func.count(User.id)))).scalar_one()
    if count > 0:
        raise HTTPException(status_code=409, detail="Konto superadmina juz istnieje.")
    user = User(
        email=body.email,
        hashed_password=hash_password(body.password),
        is_superadmin=True,
    )
    db.add(user)
    await db.commit()
    await db.refresh(user)
    return {
        "user_id": user.id,
        "email": user.email,
        "message": "Konto superadmina zostalo utworzone.",
    }
```

Note: Polish characters in error messages — use the exact strings from UI-SPEC (with proper diacritics: "Konto superadmina już istnieje." and "Konto superadmina zostało utworzone."). Read the UI-SPEC file to get exact character encoding.
